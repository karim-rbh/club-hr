function collect() {
  const sched = {};
  DAYS.forEach(d => {
    if (document.getElementById('dm_'+d.id)?.checked) {
      const s = [...document.querySelectorAll(`input[name="${d.id}_s"]:checked`)].map(x=>x.value);
      sched[d.n] = s.length ? s.join(' | ') : 'متاح';
    }
  });
  return {
    id         : Date.now(),
    ts         : new Date().toLocaleString('ar-DZ'),
    name       : document.getElementById('f_name').value.trim()||'—',
    level      : document.getElementById('f_level').value||'—',
    dept       : document.getElementById('f_dept').value||'—',
    spec       : document.getElementById('f_spec').value.trim()||'—',
    phone      : document.getElementById('f_phone').value.trim()||'—',
    insta      : document.getElementById('f_insta').value.trim()||'—',
    schedule   : Object.entries(sched).map(([d,s])=>`${d}: ${s}`).join('\n')||'لم يحدد',
    busy       : document.querySelector('input[name="busy"]:checked')?.value||'—',
    meet       : document.querySelector('input[name="meet"]:checked')?.value||'—',
    duration   : document.querySelector('input[name="dur"]:checked')?.value||'—',
    freeH      : document.querySelector('input[name="fh"]:checked')?.value||'—',
    star1      : parseInt(document.querySelector('input[name="s1"]:checked')?.value||0),
    star2      : parseInt(document.querySelector('input[name="s2"]:checked')?.value||0),
    star3      : parseInt(document.querySelector('input[name="s3"]:checked')?.value||0),
    notes      : document.getElementById('f_notes').value.trim()||'—',
    suggestions: document.getElementById('f_sugg').value.trim()||'—',
    hrInterview: document.querySelector('input[name="hrInterview"]:checked')?.value||'—',
    fields     : [...document.querySelectorAll('input[name="field"]:checked')].map(x=>x.value).join(' | ')||'—',
    fieldOther : document.getElementById('f_field_other').value.trim()||'—',
  };
}

function renderStats(data) {
  document.getElementById('st-total').textContent = data.length;

  if (data.length) {
    const avg = data.reduce((s,r)=>s+(r.star1+r.star2+r.star3)/3,0)/data.length;
    document.getElementById('st-avg').textContent = avg.toFixed(1) + '⭐';

    // Most available day
    const dayCounts = {};
    data.forEach(r => r.schedule.split('\n').forEach(line => {
      const day = line.split(':')[0]?.trim();
      if (day) dayCounts[day] = (dayCounts[day]||0) + 1;
    }));
    const topDay = Object.entries(dayCounts).sort((a,b)=>b[1]-a[1])[0];
    document.getElementById('st-day').textContent = topDay ? topDay[0].slice(0,3) : '—';

    // Most common level
    const lvlCounts = {};
    data.forEach(r=>{ lvlCounts[r.level]=(lvlCounts[r.level]||0)+1; });
    const topLvl = Object.entries(lvlCounts).sort((a,b)=>b[1]-a[1])[0];
    document.getElementById('st-level').textContent = topLvl?.[0] || '—';
  } else {
    document.getElementById('st-avg').textContent = '—';
    document.getElementById('st-day').textContent = '—';
    document.getElementById('st-level').textContent = '—';
  }
}

// ADMIN — CHARTS
// ══════════════════════════════════════
const LEVEL_COLORS = {L1:'#3b82f6',L2:'#8b5cf6',L3:'#10b981',M1:'#f59e0b',M2:'#f43f5e','دكتوراه':'#06b6d4'};
const BUSY_COLORS = ['#10b981','#3b82f6','#f59e0b','#f97316','#ef4444'];

function renderCharts(data) {
  // Levels bar chart
  const lvls = ['L1','L2','L3','M1','M2','دكتوراه'];
  const lvlCounts = {};
  data.forEach(r=>{ lvlCounts[r.level]=(lvlCounts[r.level]||0)+1; });
  const maxL = Math.max(1,...Object.values(lvlCounts));
  document.getElementById('levelChart').innerHTML = lvls.map(l=>`
    <div class="bar-item">
      <div class="bar-label">${l}</div>
      <div class="bar-track"><div class="bar-fill" style="width:${((lvlCounts[l]||0)/maxL*100)}%;background:${LEVEL_COLORS[l]||'var(--accent)'}"></div></div>
      <div class="bar-count">${lvlCounts[l]||0}</div>
    </div>`).join('');

  // Busy donut
  const busyLabels = ['😴 خفيف جداً','🙂 خفيف','😐 متوسط','😓 مرهق','🤯 مكثف جداً'];
  const busyCounts = {}; busyLabels.forEach(b=>busyCounts[b]=0);
  data.forEach(r=>{ if(busyCounts[r.busy]!==undefined) busyCounts[r.busy]++; });
  const totalBusy = data.length || 1;
  let offset = 0; const R = 35, C = 2*Math.PI*R;
  const arcs = busyLabels.map((b,i)=>{
    const pct = busyCounts[b]/totalBusy;
    const dash = pct*C; const gap = C-dash;
    const arc = `<circle cx="45" cy="45" r="${R}" fill="none" stroke="${BUSY_COLORS[i]}"
      stroke-width="14" stroke-dasharray="${dash} ${gap}"
      stroke-dashoffset="${-offset*C+C/4}" transform="rotate(-90 45 45)"/>`;
    offset += pct; return arc;
  });
  document.getElementById('busyDonut').innerHTML =
    `<circle cx="45" cy="45" r="35" fill="none" stroke="#1a2540" stroke-width="14"/>` + arcs.join('');
  document.getElementById('busyLegend').innerHTML = busyLabels.map((b,i)=>
    `<div class="dl-item"><div class="dl-dot" style="background:${BUSY_COLORS[i]}"></div>
    <span>${b.slice(2)}</span><span class="dl-val">${busyCounts[b]}</span></div>`).join('');

  // Stars summary
  if (data.length) {
    const avg = k => (data.reduce((s,r)=>s+r[k],0)/data.length).toFixed(1);
    const starStr = v => '⭐'.repeat(Math.round(parseFloat(v)));
    document.getElementById('starsSummary').innerHTML = [
      {k:'star1',l:'تنظيم النادي'},{k:'star2',l:'الأنشطة'},{k:'star3',l:'التوصية'}
    ].map(({k,l})=>`
      <div class="ss-item">
        <div class="ss-label">${l}</div>
        <div class="ss-row">
          <span class="ss-stars">${starStr(avg(k))}</span>
          <span class="ss-val">${avg(k)}/5</span>
        </div>
      </div>`).join('');
  } else {
    document.getElementById('starsSummary').innerHTML = '<p style="color:var(--txt3);font-size:12px">لا توجد بيانات بعد</p>';
  }
}

function renderTable() {
  let data = getResponses();
  const q = document.getElementById('searchInput').value.toLowerCase();
  const lf = document.getElementById('filterLevel').value;
  const mf = document.getElementById('filterMeet').value;

  if (q) data = data.filter(r=>
    r.name.toLowerCase().includes(q) ||
    r.phone.includes(q) ||
    r.spec.toLowerCase().includes(q) ||
    r.dept.toLowerCase().includes(q));
  if (lf) data = data.filter(r=>r.level===lf);
  if (mf) data = data.filter(r=>r.meet.includes(mf));

  data.sort((a,b)=>{
    const av = a[sortCol]||'', bv = b[sortCol]||'';
    return typeof av==='number' ? (av-bv)*sortDir : String(av).localeCompare(String(bv),'ar')*sortDir;
  });

  document.getElementById('shownCount').textContent = data.length;
  document.getElementById('totalCount').textContent = getResponses().length;

  const tbody = document.getElementById('tBody');
  if (!data.length) {
    tbody.innerHTML = `<tr><td colspan="12">
      <div class="empty-state"><div class="es-ico">📭</div>
      <p>لا توجد ردود مطابقة</p></div></td></tr>`;
    return;
  }

  tbody.innerHTML = data.map((r,i)=>`
    <tr onclick="showDetail('${r.id}')" style="cursor:pointer">
      <td><span class="td-name">${i+1}. ${r.name}</span></td>
      <td><span class="td-level">${r.level}</span></td>
      <td>${r.spec!=='—'?r.spec:r.dept}</td>
      <td><span style="font-family:'JetBrains Mono',monospace;font-size:11px">${r.phone}</span></td>
      <td style="max-width:120px;overflow:hidden;text-overflow:ellipsis">${r.schedule.replace(/\n/g,' • ')}</td>
      <td>${r.busy}</td>
      <td>${r.meet}</td>
      <td class="td-stars">${'⭐'.repeat(r.star1||0)} <span style="font-size:10px;color:var(--txt3)">(${r.star1}/5)</span></td>
      <td><span style="font-size:11px;color:${(r.hrInterview||'').includes('نعم')?'var(--green)':(r.hrInterview||'').includes('مجدولة')?'var(--gold)':'var(--txt3)'}">${r.hrInterview||'—'}</span></td>
      <td style="max-width:120px;overflow:hidden;text-overflow:ellipsis;font-size:11px">${r.fields||'—'}</td>
      <td style="font-size:11px;color:var(--txt3)">${r.ts}</td>
      <td onclick="event.stopPropagation()">
        <button class="td-del" onclick="deleteRow(${r.id})">حذف</button>
      </td>
    </tr>`).join('');
}

function exportExcel() {
  const data = getResponses();
  if (!data.length) { showToast('⚠️ لا توجد بيانات للتصدير'); return; }

  const rows = [['#','الاسم','المستوى','الكلية','التخصص','الهاتف','إنستغرام',
    'جدول الفراغ','الانشغال','الاجتماع','المدة','ساعات/أسبوع',
    'تقييم التنظيم','تقييم الأنشطة','التوصية','مقابلة HR','المجالات','مجال آخر',
    'التزامات','اقتراحات','تاريخ الرد']];

  data.forEach((r,i)=>rows.push([
    i+1,r.name,r.level,r.dept,r.spec,r.phone,r.insta,
    r.schedule.replace(/\n/g,' | '),r.busy,r.meet,r.duration,r.freeH,
    `${r.star1}/5`,`${r.star2}/5`,`${r.star3}/5`,r.hrInterview||'—',r.fields||'—',r.fieldOther||'—',
    r.notes,r.suggestions,r.ts
  ]));

  const ws = XLSX.utils.aoa_to_sheet(rows);
  ws['!cols'] = rows[0].map((_,i)=>({wch: i===7||i===15||i===16?30:16}));
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, 'ردود النادي');
  XLSX.writeFile(wb, `ردود_النادي_${new Date().toLocaleDateString('ar-DZ').replace(/\//g,'-')}.xlsx`);
  showToast('📥 تم تصدير Excel بنجاح!');
}
